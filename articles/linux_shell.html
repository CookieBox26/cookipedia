#!/bin/sh -eu

# 空白区切りで要素を収集
fruits="banana"
fruits=${fruits}" banana"
fruits=${fruits}" apple"
fruits=${fruits}" lemon"
fruits=${fruits}" apple"
echo ${fruits}

# 配列へ
fruits=(${fruits})
echo ${fruits[@]}

# 重複排除しソート
fruits=($(for v in "${fruits[@]}"; do echo "$v"; done | sort -u ))
echo ${fruits[@]}

# ループ処理
for fruit in "${fruits[@]}"; do
    echo $fruit
done
