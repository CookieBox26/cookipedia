from IPython.display import display, HTML
import matplotlib.colors as mcolors
import pandas as pd

div_sty = 'display: flex; align-items: flex-start; flex-wrap: wrap; gap: 2px;'
tbl_stys = [
    {'selector': '', 'props': [('border-collapse', 'collapse')]},
    {'selector': 'th,td', 'props': [
        ('padding', '0 2px'), ('background', '#ffffff'), ('font-size', 'x-small'),
        ('border', '1px solid dimgrey !important'),
        ('text-align', 'left !important')]}]
def _set_sty(row):
    return [f'background: {mcolors.to_hex(row[1])} !important', '']

html_str = ''
for group in ['BASE_COLORS', 'CSS4_COLORS', 'TABLEAU_COLORS', 'XKCD_COLORS']:
    html_str += f'
{group}
'
    colors = sorted(
        getattr(mcolors, group),
        key=lambda c: tuple(mcolors.rgb_to_hsv(mcolors.to_rgb(c))))
    df = pd.DataFrame({'name': colors})
    df.insert(0, 'sample', '')
    tbl_html_str = ''
    for i in range(0, len(colors), 20):
        tbl_html_str += df.iloc[i:(i + 20), :] \
            .style.apply(_set_sty, axis=1) \
            .set_table_styles(tbl_stys).to_html()  # .hide(axis='index')
    html_str += f'
' + tbl_html_str + '
'

display(HTML(html_str))
