import numpy as np
import pandas as pd

df_s = pd.DataFrame()
df_s['a'] = [0.9, 0.8, 0.7]
df_s['b'] = ['qqq', 'www', 'eee']
df_s['c'] = ['aaa', 'sss', 'ddd']
df_s['d'] = [10, 20, 30]
print(df_s)

n = 3
target_cols = ['a', 'b', 'c']
cols_l = [f'{col}_{i + 1}' for i in range(n) for col in target_cols]
df_l = pd.DataFrame(columns=cols_l)
df_l.loc[0] = np.concatenate([df_s.loc[i, target_cols].values for i in range(n)])
print(df_l)
