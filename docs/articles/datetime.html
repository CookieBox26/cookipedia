日付を日本語表記する


def dt_to_str(dt):
    dow = '月火水木金土日'[dt.weekday()]
    return f'{dt.month:02}月{dt.day:02}日 ({dow})'

日付を英語表記する


def dt_to_str(dt):
    dow = ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'][dt.weekday()]
    return dow + ', ' + dt.strftime('%b') + ' ' + str(dt.day).rjust(2)
